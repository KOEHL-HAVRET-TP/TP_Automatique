749__STATIC_INLINE void __NVIC_EnableIRQ(IRQn_Type IRQn)
750{
751 if ((int32_t)(IRQn) >= 0)
752 {
753 __COMPILER_BARRIER();
754 NVIC->ISER[0U] = (uint32_t)(1UL << (((uint32_t)IRQn) & 0x1FUL));
755 __COMPILER_BARRIER();
756 }
